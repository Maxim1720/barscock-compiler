program var
{identifiers}
a, b, i: int;
c, d: float;

{operators}
begin
{assignment}
a as 6;
b as 8;
{read}
read(c);
d as c plus 6;
{write}
write(d);

{id as 5 plus 5;}

{if-then-else}
if d GT c
then c as c mult 2
else c as 1.1e6:
 write(c);

{for loop}
for i as 0 to 10 do
c as c plus i:
d as d plus i;

write(c);
write(d);

while c LE d
do c as c plus 1;

end.
